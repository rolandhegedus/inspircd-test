123 template<typename T> inline std::string ConvNumeric(const T &in)
124 {
125  if (in == 0) return "0";
126  char res[MAXBUF];
127  char* out = res;
128  T quotient = in;
129  while (quotient) {
130  *out = "0123456789"[ std::abs( (long)quotient % 10 ) ];
131  ++out;
132  quotient /= 10;
133  }
134  if (in < 0)
135  *out++ = '-';
136  *out = 0;
137  std::reverse(res,out);
138  return res;
139 }
